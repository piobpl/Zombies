controller.game()
	controller.advancingStage()
		gameState.board.get()
		gameState.board.set()
			gui.board.update()
	controller.drawingStage()
		gameState.zombieDeck.get() / gameState.humanDeck.get()
		gameState.zombieHand.set() / gameState.humanHand.set()
			gui.zombieHand.update() / gui.humanHand.update() 
	controller.discardingStage()
		gui.getEventReceiver()
		eventReceiver.receive()
		gameState.zombieHand.set() / gameState.humanHand.set()
			gui.zombieHand.update() / gui.humanHand.update() 
		gui.removeEventReceiver()	
	controller.playingStage()
		gui.getEventReceiver()
		eventReceiver.receive()
		gui.removeEventReceiver()	
		Selector.getSelection(card)
			gui.getEventReceiver()
			eventReceiver.receive()
			card.isSelectionCorrect(selection)
			gui.removeEventReceiver()	
		card.makeEffect(selection, gameState)
			gui.board.update()
		gameState.zombieHand.set() / gameState.humanHand.set()
			gui.zombieHand.update() / gui.humanHand.update()
